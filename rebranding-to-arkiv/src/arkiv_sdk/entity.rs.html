7/// A generic key-value pair structure for entity annotations.
8/// Used for both string and numeric metadata attached to entities.
9#[derive(Debug, Clone, Serialize, Deserialize, RlpEncodable, RlpDecodable)]
10pub struct Annotation<T> {
11    /// The key of the annotation.
12    pub key: Key,
13    /// The value of the annotation.
14    pub value: T,
15}
16
17impl<T> Annotation<T> {
18    /// Creates a new key-value pair annotation.
19    /// Accepts any types convertible to `Key` and the annotation value.
20    pub fn new<K, V>(key: K, value: V) -> Self
21    where
22        K: Into<Key>,
23        V: Into<T>,
24    {
25        Annotation {
26            key: key.into(),
27            value: value.into(),
28        }
29    }
30}
31
32/// Type alias for string annotations (key-value pairs with `String` values).
33pub type StringAnnotation = Annotation<String>;
34
35/// Type alias for numeric annotations (key-value pairs with `u64` values).
36pub type NumericAnnotation = Annotation<u64>;
37
38/// A type alias for the hash used to identify entities in Arkiv.
39pub type Hash = B256;

41/// Type alias for the key used in annotations.
42pub type Key = String;

44/// Type representing a create transaction in Arkiv.
45/// Used to define new entities, including their data, BTL, and annotations.
46#[derive(Debug, Clone, Default, RlpEncodable, RlpDecodable, Deserialize)]
47#[rlp(trailing)]
48pub struct Create {
49    /// The block-to-live (BTL) for the entity.
50    pub btl: u64,
51    /// The data associated with the entity.
52    pub data: Bytes,
53    /// String annotations for the entity.
54    pub string_annotations: Vec<StringAnnotation>,
55    /// Numeric annotations for the entity.
56    pub numeric_annotations: Vec<NumericAnnotation>,
57}
58
59/// Type representing an update transaction in Arkiv.
60/// Used to update existing entities, including their data, BTL, and annotations.
61#[derive(Debug, Clone, Default, RlpEncodable, RlpDecodable, Deserialize)]
62#[rlp(trailing)]
63pub struct Update {
64    /// The key of the entity to update.
65    pub entity_key: Hash,
66    /// The updated block-to-live (BTL) for the entity.
67    pub btl: u64,
68    /// The updated data for the entity.
69    pub data: Bytes,
70    /// Updated string annotations for the entity.
71    pub string_annotations: Vec<StringAnnotation>,
72    /// Updated numeric annotations for the entity.
73    pub numeric_annotations: Vec<NumericAnnotation>,
74}

76/// Type alias for a delete operation (just the entity key).
77pub type ArkivDelete = Hash;

79/// Type representing an extend transaction in Arkiv.
80/// Used to extend the BTL of an entity by a number of blocks.
81#[derive(Debug, Clone, Default, RlpEncodable, RlpDecodable, Deserialize)]
82pub struct Extend {
83    /// The key of the entity to extend.
84    pub entity_key: Hash,
85    /// The number of blocks to extend the BTL by.
86    pub number_of_blocks: u64,
87}

89/// Type representing a transaction in Arkiv, including creates, updates, deletes, and extensions.
90/// Used as the main payload for submitting entity changes to the chain.
91#[derive(Debug, Clone, Default, RlpEncodable, RlpDecodable)]
92pub struct ArkivTransaction {
93    /// A list of entities to create.
94    pub creates: Vec<Create>,
95    /// A list of entities to update.
96    pub updates: Vec<Update>,
97    /// A list of entity keys to delete.
98    pub deletes: Vec<ArkivDelete>,
99    /// A list of entities to extend.
100    pub extensions: Vec<Extend>,
101}

103/// Represents an entity with data, BTL, and annotations.
104/// Used for reading entity state from the chain.
105#[derive(Debug, Clone, Default, RlpEncodable, RlpDecodable, Serialize, Deserialize)]
106pub struct Entity {
107    /// The data associated with the entity.
108    pub data: String,
109    /// The block-to-live (BTL) for the entity.
110    pub btl: u64,
111    /// String annotations for the entity.
112    pub string_annotations: Vec<StringAnnotation>,
113    /// Numeric annotations for the entity.
114    pub numeric_annotations: Vec<NumericAnnotation>,
115}

117/// Represents the result of creating or updating an entity.
118/// Contains the entity key and its expiration block.
119#[derive(Debug, Clone, Default, RlpEncodable, RlpDecodable, Serialize)]
120pub struct EntityResult {
121    /// The key of the entity.
122    pub entity_key: Hash,
123    /// The block number at which the entity expires.
124    pub expiration_block: u64,
125}
126
127/// Represents the result of extending an entity's BTL.
128/// Contains the entity key, old expiration block, and new expiration block.
129#[derive(Debug)]
130pub struct ExtendResult {
131    /// The key of the entity.
132    pub entity_key: Hash,
133    /// The old expiration block of the entity.
134    pub old_expiration_block: u64,
135    /// The new expiration block of the entity.
136    pub new_expiration_block: u64,
137}

147impl Create {
148    /// Creates a new `Create` operation with empty annotations.
149    /// Accepts a payload as bytes and a BTL value.
150    pub fn new(payload: Vec<u8>, btl: u64) -> Self {
151        Self {
152            btl,
153            data: Bytes::from(payload),
154            string_annotations: Vec::new(),
155            numeric_annotations: Vec::new(),
156        }
157    }
158
159    /// Creates a new `Create` request from any type that can be converted to `String`.
160    pub fn from_string(payload: impl Into<String>, btl: u64) -> Self {
161        Self {
162            btl,
163            data: Bytes::from(payload.into().into_bytes()),
164            string_annotations: Vec::new(),
165            numeric_annotations: Vec::new(),
166        }
167    }
168
169    /// Adds a string annotation to the entity.
170    /// Returns the modified `Create` for chaining.
171    pub fn annotate_string(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
172        self.string_annotations.push(Annotation {
173            key: key.into(),
174            value: value.into(),
175        });
176        self
177    }
178
179    /// Adds a numeric annotation to the entity.
180    /// Returns the modified `Create` for chaining.
181    pub fn annotate_number(mut self, key: impl Into<String>, value: u64) -> Self {
182        self.numeric_annotations.push(Annotation {
183            key: key.into(),
184            value,
185        });
186        self
187    }
188}
