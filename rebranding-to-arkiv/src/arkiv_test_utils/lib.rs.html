19/// Initializes the logger for tests
20pub fn init_logger(should_init: bool) {
21    // Check if TEST_ENABLE_ALL_LOGS_OVERRIDE environment variable overrides the should_init parameter
22    let env_override = env::var("TEST_ENABLE_ALL_LOGS_OVERRIDE")
23        .map(|val| val.parse::<bool>().unwrap_or(false))
24        .unwrap_or(false);
25    let should_enable = should_init || env_override;
26
27    if should_enable {
28        if let Ok(_env) = env::var("RUST_LOG") {
29            env_logger::try_init_from_env(env_logger::Env::default()).ok();
30        } else {
31            env_logger::builder()
32                .filter_level(log::LevelFilter::Debug)
33                .format_timestamp(Some(env_logger::TimestampPrecision::Millis))
34                .try_init()
35                .ok();
36        }
37    }
38}

40/// Removes all existing entities from the Arkiv node
41pub async fn cleanup_entities(client: &ArkivClient, account: Address) -> Result<()> {
42    let all_entity_keys = client.get_all_entity_keys().await?;
43    log::info!("Removing all existing entities: {:?}", all_entity_keys);
44    if !all_entity_keys.is_empty() {
45        client.remove_entries(account, all_entity_keys).await?;
46    }
47    Ok(())
48}

60/// Finds the transaction that created an entry by searching backwards through blocks.
61/// Returns (transaction_hash, block_number) or None if not found.
62pub async fn find_entry_creation_transaction(
63    client: &ArkivClient,
64    entry_id: Hash,
65) -> Result<Option<(B256, u64)>> {
66    let current_block = client.get_current_block_number().await?;
67    for block_number in (1..=current_block).rev() {
68        let block = client
69            .get_rpc_client()
70            .get_block_by_number(BlockNumberOrTag::Number(block_number))
71            .await?
72            .ok_or_else(|| anyhow::anyhow!("Block {} not found", block_number))?;
73
74        for tx in block.transactions.hashes() {
75            if let Ok(Some(receipt)) = client.get_rpc_client().get_transaction_receipt(tx).await {
76                if let Ok(log_entry_id) = ArkivClient::extract_entity_id(receipt.logs()) {
77                    if log_entry_id == entry_id {
78                        return Ok(Some((tx, block_number)));
79                    }
80                }
81            }
82        }
83    }
84    Ok(None)
85}
