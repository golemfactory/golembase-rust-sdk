15/// Represents a GolemBase entity
16#[derive(Clone, Debug, Serialize, Deserialize)]
17pub struct Entity {
18    #[serde(serialize_with = "serialize_b256")]
19    pub key: B256,
20    #[serde(serialize_with = "serialize_base64")]
21    pub data: Bytes,
22    pub btl: u64,
23    #[serde(serialize_with = "serialize_address")]
24    pub owner: Address,
25    pub expires_at: Option<u64>,
26    pub string_annotations: Vec<StringAnnotation>,
27    pub numeric_annotations: Vec<NumericAnnotation>,
28}

30impl Entity {
31    /// Create a new entity from a GolemBase Create object and owner
32    pub fn create(create: Create, owner: Address) -> Self {
33        let entity = Self {
34            key: B256::ZERO, // Temporary key, will be calculated below
35            data: create.data.clone(),
36            btl: create.btl,
37            owner,
38            expires_at: None, // Will be computed based on current block + BTL
39            string_annotations: create.string_annotations,
40            numeric_annotations: create.numeric_annotations,
41        };
42
43        entity
44    }
45
46    /// Calculate the hash of this entity based on its content, owner, block number, index and transaction hash
47    /// This creates a deterministic identifier that combines entity content with block context
48    pub fn calculate_hash(&self, block_number: u64, idx: usize, transaction_hash: B256) -> B256 {
49        let mut content_bytes = Vec::new();
50        content_bytes.extend_from_slice(&self.data);
51        content_bytes.extend_from_slice(self.btl.to_le_bytes().as_slice());
52        content_bytes.extend_from_slice(self.owner.as_slice());
53        content_bytes.extend_from_slice(block_number.to_le_bytes().as_slice());
54        content_bytes.extend_from_slice(idx.to_le_bytes().as_slice());
55        content_bytes.extend_from_slice(transaction_hash.as_slice());
56
57        // Annotations don't need to be part of hash, because it will be unique anyway.
58
59        keccak256(&content_bytes)
60    }
61
62    /// Set the entity key to a hash based on block number, index and transaction hash
63    /// Also sets the expiration block number based on current block + BTL
64    /// This modifies the entity in place and returns self for chaining
65    pub fn with_hash(mut self, block_number: u64, idx: usize, transaction_hash: B256) -> Self {
66        self.key = self.calculate_hash(block_number, idx, transaction_hash);
67        self.expires_at = Some(block_number + self.btl);
68        self
69    }
70
71    /// Update this entity with data from a GolemBase Update object
72    /// This modifies the entity in place and returns self for chaining
73    pub fn update(&mut self, update: &Update) -> &mut Self {
74        self.data = update.data.clone();
75        self.btl = update.btl;
76        self.string_annotations = update.string_annotations.clone();
77        self.numeric_annotations = update.numeric_annotations.clone();
78        self
79    }

81    /// Check if this entity matches a specific query condition
82    pub fn matches_condition(&self, condition: &QueryCondition) -> bool {
83        match condition {
84            QueryCondition::StringEquals(key, value) => {
85                // Check if the entity has this string annotation key with the specified value
86                self.string_annotations
87                    .iter()
88                    .find(|a| a.key == *key)
89                    .map_or(false, |a| a.value == *value)
90            }
91            QueryCondition::NumericEquals(key, value) => {
92                // Check if the entity has this numeric annotation key with the specified value
93                self.numeric_annotations
94                    .iter()
95                    .find(|a| a.key == *key)
96                    .map_or(false, |a| a.value == *value)
97            }
98            QueryCondition::OwnerEquals(value) => {
99                // Compare the entity's owner with the specified address
100                self.owner == *value
101            }
102        }
103    }
104}

140/// Internal state of the entity database
141#[derive(Clone, Debug, Default)]
142struct EntityDbState {
143    entities: HashMap<B256, Entity>,
144    string_annotations: HashMap<String, Vec<B256>>,
145    numeric_annotations: HashMap<String, Vec<B256>>, // Indexed by key, not value
146    entities_by_owner: HashMap<Address, Vec<B256>>,  // Index entities by owner
147}
148
149impl EntityDbState {
150    /// Evaluate a parsed expression to get matching entity keys
151    fn evaluate_expression(&self, expression: &Expression) -> Vec<B256> {
152        match expression {
153            Expression::Condition(condition) => {
154                // Use annotation maps to find entity keys, then filter by actual annotation values
155                self.get_candidates(condition)
156                    .into_iter()
157                    .filter_map(|key| {
158                        self.entities.get(&key).and_then(|entity| {
159                            if entity.matches_condition(condition) {
160                                Some(key)
161                            } else {
162                                None
163                            }
164                        })
165                    })
166                    .collect()
167            }
168            Expression::And(left, right) => {
169                let left_keys: HashSet<B256> = self.evaluate_expression(left).into_iter().collect();
170                let right_keys: HashSet<B256> =
171                    self.evaluate_expression(right).into_iter().collect();
172                left_keys.intersection(&right_keys).cloned().collect()
173            }
174            Expression::Or(left, right) => {
175                let left_keys: HashSet<B256> = self.evaluate_expression(left).into_iter().collect();
176                let right_keys: HashSet<B256> =
177                    self.evaluate_expression(right).into_iter().collect();
178                left_keys.union(&right_keys).cloned().collect()
179            }
180        }
181    }
182
183    /// Get candidate entity keys from annotation maps for a given condition
184    fn get_candidates(&self, condition: &QueryCondition) -> Vec<B256> {
185        match condition {
186            QueryCondition::StringEquals(key, _) => {
187                // Find entities that have this string annotation key
188                self.string_annotations
189                    .get(key)
190                    .cloned()
191                    .unwrap_or_default()
192            }
193            QueryCondition::NumericEquals(key, _) => {
194                // Find entities that have this numeric annotation key
195                self.numeric_annotations
196                    .get(key)
197                    .cloned()
198                    .unwrap_or_default()
199            }
200            QueryCondition::OwnerEquals(value) => {
201                // Find entities by owner address
202                self.entities_by_owner
203                    .get(value)
204                    .cloned()
205                    .unwrap_or_default()
206            }
207        }
208    }
209
210    /// Query entity keys by parsing a query string and finding keys matching all specified annotations
211    /// Query format: Supports &&, ||, parentheses, and meta-annotations
212    /// Examples: "test_type = \"Test\"", "priority = 1", "tag = \"important\" && priority = 1"
213    fn query_entity_keys(&self, query: &str) -> Result<Vec<B256>> {
214        // Parse query string to extract conditions
215        let expression =
216            Parser::parse_query(query).map_err(|e| anyhow!("Failed to parse query: {}", e))?;
217
218        // Evaluate the expression to get matching entity keys
219        Ok(self.evaluate_expression(&expression))
220    }
221
222    /// Query entities by parsing a query string and finding entities matching all specified annotations
223    /// Query format: Supports &&, ||, parentheses, and meta-annotations
224    /// Examples: "test_type = \"Test\"", "priority = 1", "tag = \"important\" && priority = 1"
225    fn query_entities(&self, query: &str) -> Result<Vec<Entity>> {
226        let matching_keys = self.query_entity_keys(query)?;
227
228        // Return entities for matching keys
229        let mut entities = Vec::new();
230        for key in matching_keys {
231            if let Some(entity) = self.entities.get(&key) {
232                entities.push(entity.clone());
233            }
234        }
235
236        Ok(entities)
237    }
238}

250            state: Arc::new(RwLock::new(EntityDbState {
251                entities: HashMap::new(),
252                string_annotations: HashMap::new(),
253                numeric_annotations: HashMap::new(),
254                entities_by_owner: HashMap::new(),

259    /// Add an entity to the database
260    pub async fn add_entity(&self, entity: Entity) {
261        let key = entity.key;
262        let mut state = self.state.write().await;
263
264        log::info!("Creating entity: key={}", key);
265        // Add to main entities map
266        state.entities.insert(key, entity.clone());
267
268        // Update owner index
269        state
270            .entities_by_owner
271            .entry(entity.owner)
272            .or_insert_with(Vec::new)
273            .push(entity.key);
274
275        // Update annotation indices
276        Self::update_annotations(
277            &mut state,
278            entity.key,
279            &entity.string_annotations,
280            &entity.numeric_annotations,
281        );
282    }

284    /// Update an existing entity in the database with new data
285    /// Returns true if entity was updated, false if entity doesn't exist
286    pub async fn update_entity(&self, entity_key: &B256, update: &Update) -> bool {
287        let mut state = self.state.write().await;
288
289        if let Some(entity) = state.entities.get_mut(entity_key) {
290            log::info!("Updating entity: key={}", entity_key);
291            // Update the entity using the existing update method
292            entity.update(update);
293
294            // Get entity data to avoid borrowing conflicts
295            let entity_key = entity.key;
296            let new_string_annotations = entity.string_annotations.clone();
297            let new_numeric_annotations = entity.numeric_annotations.clone();
298
299            // Update annotation indices
300            Self::update_annotations(
301                &mut state,
302                entity_key,
303                &new_string_annotations,
304                &new_numeric_annotations,
305            );
306
307            true
308        } else {
309            // Entity doesn't exist, ignore the update
310            false
311        }
312    }

323    /// Function to remove all annotations for a specific entity
324    fn remove_entity_annotations(state: &mut EntityDbState, entity_key: B256) {
325        // Remove all existing annotations for this entity from hash maps
326        for (_, keys) in state.string_annotations.iter_mut() {
327            keys.retain(|&k| k != entity_key);
328        }
329        for (_, keys) in state.numeric_annotations.iter_mut() {
330            keys.retain(|&k| k != entity_key);
331        }
332    }
333
334    /// Unified function to update annotation indices
335    /// Removes all existing annotations for the entity and adds new ones
336    fn update_annotations(
337        state: &mut EntityDbState,
338        entity_key: B256,
339        new_string_annotations: &[StringAnnotation],
340        new_numeric_annotations: &[NumericAnnotation],
341    ) {
342        // Remove all existing annotations first
343        Self::remove_entity_annotations(state, entity_key);
344
345        // Add new annotations to hash maps
346        for annotation in new_string_annotations {
347            state
348                .string_annotations
349                .entry(annotation.key.clone())
350                .or_insert_with(Vec::new)
351                .push(entity_key);
352        }
353        for annotation in new_numeric_annotations {
354            state
355                .numeric_annotations
356                .entry(annotation.key.clone())
357                .or_insert_with(Vec::new)
358                .push(entity_key);
359        }
360    }

405    /// Remove an entity from the database
406    pub async fn remove_entity(&self, key: &B256) -> Option<Entity> {
407        let mut state = self.state.write().await;
408        if let Some(entity) = state.entities.remove(key) {
409            log::info!("Removing entity: key={}, owner={}", key, entity.owner);
410            // Remove from owner index
411            if let Some(keys) = state.entities_by_owner.get_mut(&entity.owner) {
412                keys.retain(|&k| k != entity.key);
413            }
414
415            // Remove all annotations for this entity
416            Self::remove_entity_annotations(&mut state, entity.key);
417            Some(entity)
418        } else {
419            None
420        }
421    }
