19/// Event structure that stores all information needed to recreate an alloy Log
20#[derive(Debug, Clone)]
21pub struct LogEvent {
22    /// The contract address that emitted the event
23    pub address: Address,
24    /// The event topic (hash of the event signature)
25    pub topic: B256,
26    /// The entity ID (first indexed parameter)
27    pub entity_id: B256,
28    /// Additional data (second indexed parameter for some events)
29    pub additional_data: Option<B256>,
30    /// The block number where the event occurred
31    pub block_number: u64,
32    /// The transaction hash that triggered the event
33    pub transaction_hash: B256,
34    /// The transaction index within the block
35    pub transaction_index: u64,
36    /// The log index within the transaction
37    pub log_index: u64,
38    /// The block hash
39    pub block_hash: B256,
40}
41
42impl Into<Log> for LogEvent {
43    fn into(self) -> Log {
44        let mut topics = vec![self.topic];
45        topics.push(self.entity_id);
46        if let Some(data) = self.additional_data {
47            topics.push(data);
48        }
49
50        let inner =
51            alloy::primitives::Log::new(self.address, topics, alloy::primitives::Bytes::new())
52                .unwrap_or_else(|| alloy::primitives::Log::empty());
53
54        Log {
55            inner,
56            block_timestamp: None, // We don't have timestamp in our LogEvent
57            block_number: Some(self.block_number),
58            transaction_hash: Some(self.transaction_hash),
59            transaction_index: Some(self.transaction_index),
60            log_index: Some(self.log_index),
61            block_hash: Some(self.block_hash),
62            removed: false,
63        }
64    }
65}

69pub struct EventFilter {

74    /// Block range to filter by

89/// Information about a subscription including its filter
90#[derive(Clone)]
91struct SubscriptionInfo {
92    sender: broadcast::Sender<LogEvent>,
93    filter: Option<EventFilter>,
94}
95
96impl SubscriptionInfo {
97    /// Send a log event to the subscription with logging
98    fn emit_event(&self, log_event: &LogEvent) {
99        let _ = self.sender.send(log_event.clone());
100        log::info!(
101            "Sent event to subscription: block={}, tx={}, log_index={}, entity_id={}",
102            log_event.block_number,
103            log_event.transaction_hash,
104            log_event.log_index,
105            log_event.entity_id
106        );
107    }
108}

110/// Inner state of the EventEmitter, wrapped in a single RwLock
111#[derive(Default)]
112struct EventEmitterState {
113    subscriptions: HashMap<String, SubscriptionInfo>,
114    /// Pending logs per block, waiting to be emitted with proper indices
115    pending_logs: HashMap<u64, Vec<LogEvent>>,
116}

118/// Event emission system for GolemBase mock
119#[derive(Clone, Default)]
120pub struct EventEmitter {
121    state: Arc<RwLock<EventEmitterState>>,
122}
123
124impl EventEmitter {
125    /// Create a new event emitter
126    pub fn new() -> Self {
127        Self::default()
128    }
129
130    /// Generate a unique subscription ID
131    fn generate_subscription_id() -> String {
132        format!(
133            "events_{}",
134            SystemTime::now()
135                .duration_since(UNIX_EPOCH)
136                .unwrap()
137                .as_nanos()
138        )
139    }
140
141    /// Get the latest block number from the state
142    async fn get_latest_block_number(&self) -> u64 {
143        let state = self.state.read().await;
144        state.pending_logs.keys().max().copied().unwrap_or(0)
145    }
146
147    /// Extract block range from Filter
148    async fn extract_block_range(
149        &self,
150        filter: &Filter,
151    ) -> anyhow::Result<(Option<u64>, Option<u64>)> {
152        let latest = self.get_latest_block_number().await;
153
154        match &filter.block_option {
155            FilterBlockOption::Range {
156                from_block,
157                to_block,
158            } => {
159                let from = from_block.and_then(|b| self.extract_block_number(&b, latest));
160                let to = to_block.and_then(|b| self.extract_block_number(&b, latest));
161                Ok((from, to))
162            }
163            FilterBlockOption::AtBlockHash(_) => {
164                // Single block hash - we can't resolve the hash to a block number
165                anyhow::bail!("[Not implemented] Cannot return events for a block hash")
166            }
167        }
168    }
169
170    /// Extract block number from BlockNumberOrTag
171    fn extract_block_number(
172        &self,
173        block_tag: &BlockNumberOrTag,
174        latest_block_number: u64,
175    ) -> Option<u64> {
176        match block_tag {
177            BlockNumberOrTag::Number(n) => Some(*n),
178            BlockNumberOrTag::Earliest => Some(0),
179            // Latest, Finalized, Safe all map to the current latest block number
180            // since we treat them as equivalent in our mock implementation
181            BlockNumberOrTag::Latest | BlockNumberOrTag::Finalized | BlockNumberOrTag::Safe => {
182                Some(latest_block_number)
183            }
184            // Pending is relative to current state - we can't extract specific numbers
185            BlockNumberOrTag::Pending => None,
186        }
187    }
188
189    /// Add a log event to the pending logs for a block
190    async fn add_pending_log(&self, mut log_event: LogEvent) {
191        let mut state = self.state.write().await;
192        let block_logs = state
193            .pending_logs
194            .entry(log_event.block_number)
195            .or_insert_with(Vec::new);
196
197        // Set log index to the current position in the vector
198        log_event.log_index = block_logs.len() as u64;
199        block_logs.push(log_event);
200    }
201
202    /// Emit a log event to all subscribers, applying their individual filters
203    async fn emit_log_event(&self, log_event: LogEvent) {
204        let state = self.state.read().await;
205        for (_, subscription_info) in state.subscriptions.iter() {
206            // Check if event matches the subscription's filter
207            let should_send = match &subscription_info.filter {
208                Some(filter) => self.log_event_matches_filter(&log_event, filter),
209                None => true,
210            };
211
212            if should_send {
213                subscription_info.emit_event(&log_event);
214            }
215        }
216    }
217
218    /// Subscribe to events with an optional filter
219    pub async fn subscribe_to_events(
220        &self,
221        filter: Option<Filter>,
222    ) -> anyhow::Result<broadcast::Receiver<LogEvent>> {
223        let subscription_id = Self::generate_subscription_id();
224        let (sender, receiver) = broadcast::channel(1000); // Buffer up to 1000 events
225
226        let event_filter = match filter {
227            Some(f) => {
228                let (from_block, to_block) = self.extract_block_range(&f).await?;
229                Some(EventFilter {
230                    addresses: f.address.clone(),
231                    topics: f.topics.iter().cloned().collect(),
232                    from_block,
233                    to_block,
234                })
235            }
236            None => None,
237        };
238
239        let subscription_info = SubscriptionInfo {
240            sender: sender.clone(),
241            filter: event_filter.clone(),
242        };
243
244        let mut state = self.state.write().await;
245        state
246            .subscriptions
247            .insert(subscription_id, subscription_info.clone());
248
249        log::info!("Created subscription for events with filter: {event_filter:?}");
250
251        // Emit events from past blocks if they match the filter
252        if let Some(filter) = &event_filter {
253            log::info!("Emitting events from past blocks using filter: {filter:?}");
254            self.emit_past_blocks(&subscription_info, filter, &state)
255                .await;
256        }
257
258        Ok(receiver)
259    }
260
261    /// Emit events from past blocks that match the filter
262    async fn emit_past_blocks(
263        &self,
264        subscription_info: &SubscriptionInfo,
265        filter: &EventFilter,
266        state: &EventEmitterState,
267    ) {
268        for (_, logs) in &state.pending_logs {
269            // Emit logs that match the filter (which includes block range)
270            for log_event in logs {
271                if self.log_event_matches_filter(log_event, filter) {
272                    subscription_info.emit_event(log_event);
273                }
274            }
275        }
276    }
277
278    /// Check if a log event matches the given filter
279    fn log_event_matches_filter(&self, log_event: &LogEvent, filter: &EventFilter) -> bool {
280        // Check block range first
281        if let Some(from_block) = filter.from_block {
282            if log_event.block_number < from_block {
283                return false;
284            }
285        }
286        if let Some(to_block) = filter.to_block {
287            if log_event.block_number > to_block {
288                return false;
289            }
290        }
291
292        // Check if filter has any address restrictions
293        if !filter.addresses.is_empty() {
294            // Check if the log event address matches any of the filter addresses
295            let address_matches = filter.addresses.contains(&log_event.address);
296            if !address_matches {
297                return false;
298            }
299        }
300
301        // Check if filter has any topic restrictions
302        if !filter.topics.is_empty() {
303            // Check if the event topic matches any of the filter topics
304            // Topics is a Vec<FilterSet<B256>>, so we need to check if any FilterSet contains the topic
305            let topic_matches = filter
306                .topics
307                .iter()
308                .any(|topic_set| topic_set.contains(&log_event.topic));
309            if !topic_matches {
310                return false;
311            }
312        }
313
314        true
315    }
316}

319impl EntityEventHandler for EventEmitter {
320    async fn on_entity_created(&self, entity: &Entity, block: &Block, transaction: &Transaction) {
321        let transaction_index = block.find_transaction_index(&transaction.hash).unwrap_or(0);
322
323        let log_event = LogEvent {
324            address: GOLEM_BASE_STORAGE_PROCESSOR_ADDRESS,
325            topic: golem_base_storage_entity_created(),
326            entity_id: entity.key,
327            additional_data: None,
328            block_number: block.header.block_number,
329            transaction_hash: transaction.hash,
330            transaction_index,
331            log_index: 0, // Will be set correctly in add_pending_log
332            block_hash: block.header.block_hash,
333        };
334
335        self.add_pending_log(log_event).await;
336    }
337

373
374    async fn finish_block(&self, block_number: u64) {
375        let logs = {
376            let state = self.state.read().await;
377            state.pending_logs.get(&block_number).cloned()
378        };
379
380        if let Some(logs) = logs {
381            // Emit all logs in the order they were added (already correct order)
382            for log_event in logs {
383                self.emit_log_event(log_event).await;
384            }
385        }
386    }
