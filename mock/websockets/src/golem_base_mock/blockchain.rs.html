77    /// Validate transaction nonce
78    pub async fn validate_transaction_nonce(
79        &self,
80        transaction: &Transaction,
81    ) -> anyhow::Result<()> {
82        let state = self.state.read().await;
83        let sender_account = state.accounts.get(&transaction.from);
84
85        let expected_nonce = if let Some(account) = sender_account {
86            account.nonce
87        } else {
88            U256::ZERO
89        };
90
91        if U256::from(transaction.nonce) < expected_nonce {
92            return Err(anyhow::anyhow!(
93                "nonce too low: next nonce {}, tx nonce {}",
94                expected_nonce,
95                transaction.nonce
96            ));
97        }
98
99        Ok(())
100    }

102    /// Add a block to the blockchain
103    pub async fn add_block(&self, block: Block) {
104        let mut state = self.state.write().await;
105        let block_number = block.header.block_number;
106        let block_hash = block.header.block_hash;
107
108        // Process all transactions in the block
109        let mut all_logs = Vec::new();
110        for transaction in &block.transactions {
111            let transaction_hash = transaction.hash;
112
113            // Add transaction to transactions map
114            state
115                .transactions
116                .insert(transaction_hash, transaction.clone());
117
118            // Update accounts
119            Self::update_account_for_transaction(&mut state, &transaction);
120
121            // Extract and add entities from transaction data, collect logs
122            let transaction_logs = self
123                .extract_entity_from_transaction(transaction, &block)
124                .await;
125            all_logs.extend(transaction_logs);
126        }
127
128        // Create a new block with logs and add it to the state
129        let block = Block {
130            header: block.header.clone(),
131            transactions: block.transactions.clone(),
132            transaction_logs: all_logs,
133        };
134
135        let block = Arc::new(block);
136        state.blocks_by_number.insert(block_number, block.clone());
137        state.blocks_by_hash.insert(block_hash, block.clone());
138
139        // Finish processing the block and emit all collected events.
140        // At this point Blockchain must be ready to return block if caller asks,
141        // so emit events outside of lock and never move this earlier
142        drop(state);
143        self.event_handler.finish_block(block_number).await;
144    }

146    /// Update account state based on a transaction
147    fn update_account_for_transaction(state: &mut BlockchainState, transaction: &Arc<Transaction>) {
148        // Update sender account
149        let sender = transaction.from;
150        let sender_account = state
151            .accounts
152            .entry(sender)
153            .or_insert_with(|| Account::new(sender));
154        sender_account.nonce += U256::from(1);
155        sender_account.balance = sender_account.balance.saturating_sub(transaction.value);
156        sender_account.transactions.push(transaction.hash);
157
158        // Update receiver account
159        let receiver = transaction.to;
160        let receiver_account = state
161            .accounts
162            .entry(receiver)
163            .or_insert_with(|| Account::new(receiver));
164        receiver_account.balance += transaction.value;
165        receiver_account
166            .received_transactions
167            .push(transaction.hash);
168
169        if transaction.value > U256::ZERO {
170            log::debug!(
171                "Account transfer: {} -> {} (value: {} ETH, tx: 0x{:x})",
172                transaction.from,
173                transaction.to,
174                wei_to_eth(transaction.value),
175                transaction.hash
176            );
177        }
178    }

180    /// Extract entity from transaction data and modify entity database state
181    /// Checks if transaction is to storage contract and decodes GolemBase entity operations
182    /// Returns transaction logs for the block
183    async fn extract_entity_from_transaction(
184        &self,
185        transaction: &Arc<Transaction>,
186        block: &Block,
187    ) -> Vec<TransactionLog> {
188        let mut logs = Vec::new();
189
190        // Check if transaction is to the GolemBase storage processor contract
191        if transaction.to != GOLEM_BASE_STORAGE_PROCESSOR_ADDRESS {
192            return logs;
193        }
194
195        // Try to decode the transaction data as a GolemBaseTransaction
196        // This is the inverse of the encoding shown in send_db_transaction
197        match GolemBaseTransaction::decode(&mut transaction.data.as_ref()) {
198            Ok(golem_tx) => {
199                // Process creates
200                for (idx, create) in golem_tx.creates.into_iter().enumerate() {
201                    let entity = Entity::create(create, transaction.from).with_hash(
202                        block.header.block_number,
203                        idx,
204                        transaction.hash,
205                    );
206                    self.entity_db.add_entity(entity.clone()).await;
207
208                    // Create log for entity creation
209                    let create_log = TransactionLog::create_entity_log(
210                        transaction,
211                        golem_base_storage_entity_created(),
212                        entity.key,
213                    );
214                    logs.push(create_log);
215
216                    // Emit event
217                    self.event_handler
218                        .on_entity_created(&entity, &block, transaction)
219                        .await;
220
221                    log::info!(
222                        "Entity created: 0x{:x}, owner: 0x{:x}, tx: 0x{:x}",
223                        entity.key,
224                        entity.owner,
225                        transaction.hash
226                    );
227                }
228
229                // Process updates
230                for update in &golem_tx.updates {
231                    let entity_key = update.entity_key;
232
233                    // Update the entity directly in the database
234                    self.entity_db.update_entity(&entity_key, update).await;
235
236                    // Create log for entity update
237                    let update_log = TransactionLog::create_entity_log(
238                        transaction,
239                        golem_base_storage_entity_updated(),
240                        entity_key,
241                    );
242                    logs.push(update_log);
243
244                    // Emit event
245                    if let Some(entity) = self.entity_db.get_entity(&entity_key).await {
246                        self.event_handler
247                            .on_entity_updated(&entity, &block, transaction)
248                            .await;
249                    }
250
251                    log::info!(
252                        "Entity updated: 0x{:x}, tx: 0x{:x}",
253                        entity_key,
254                        transaction.hash
255                    );
256                }
257
258                // Process extensions
259                for extend in &golem_tx.extensions {
260                    let entity_key = extend.entity_key;
261                    let number_of_blocks = extend.number_of_blocks;
262
263                    // For extensions, we need to update the existing entity's BTL
264                    self.entity_db
265                        .update_entity_btl(&entity_key, number_of_blocks)
266                        .await;
267
268                    // Create log for entity extension
269                    let extend_log = TransactionLog::create_entity_log(
270                        transaction,
271                        golem_base_storage_entity_ttl_extended(),
272                        entity_key,
273                    );
274                    logs.push(extend_log);
275
276                    log::info!(
277                        "Entity extended: 0x{:x}, new BTL: {}, tx: 0x{:x}",
278                        entity_key,
279                        number_of_blocks,
280                        transaction.hash
281                    );
282                }
283
284                // Process deletes
285                for delete in &golem_tx.deletes {
286                    let key = *delete;
287                    if let Some(entity) = self.entity_db.remove_entity(&key).await {
288                        // Create log for entity deletion
289                        let delete_log = TransactionLog::create_entity_log(
290                            transaction,
291                            golem_base_storage_entity_deleted(),
292                            key,
293                        );
294                        logs.push(delete_log);
295
296                        // Emit event
297                        self.event_handler
298                            .on_entity_removed(&entity, &block, transaction)
299                            .await;
300
301                        log::info!(
302                            "Entity deleted: 0x{:x}, owner: 0x{:x}, tx: 0x{:x}",
303                            entity.key,
304                            entity.owner,
305                            transaction.hash
306                        );
307                    } else {
308                        log::warn!(
309                            "Entity not found for deletion: 0x{:x}, tx: 0x{:x}",
310                            key,
311                            transaction.hash
312                        );
313                    }
314                }
315            }
316            Err(e) => {
317                log::warn!("Failed to decode GolemBase transaction: {}", e);
318            }
319        }
320
321        logs
322    }

354    /// Find the block that contains a specific transaction
355    pub async fn find_block_containing_transaction(
356        &self,
357        transaction_hash: &B256,
358    ) -> Option<Arc<Block>> {
359        let state = self.state.read().await;
360        for (_, block) in &state.blocks_by_number {
361            for tx in &block.transactions {
362                if tx.hash == *transaction_hash {
363                    return Some(block.clone());
364                }
365            }
366        }
367        None
368    }

448    /// Create and add genesis block
449    pub async fn create_genesis_block(&self) {
450        let genesis_block = Block::new(
451            0,                      // Genesis block number
452            B256::ZERO,             // No previous block
453            Vec::new(),             // No transactions
454            U256::from(30_000_000), // Gas limit
455            U256::ZERO,             // No gas used
456            SystemTime::now()
457                .duration_since(UNIX_EPOCH)
458                .unwrap_or_default()
459                .as_secs(),
460        );
461
462        self.add_block(genesis_block).await;
463    }
