9pub struct DisplayEnabler<'a, Type>(pub &'a Type);
10
11pub trait EnableDisplay<Type> {
12    fn display(&self) -> DisplayEnabler<Type>;
13}
14
15impl<Type> EnableDisplay<Type> for Type {
16    fn display(&self) -> DisplayEnabler<Type> {
17        DisplayEnabler(self)
18    }
19}

21impl<Type> std::fmt::Display for DisplayEnabler<'_, Option<Type>>
22where
23    Type: std::fmt::Display,
24{
25    fn fmt(&self, f: &mut Formatter<'_>) -> Result<(), Error> {
26        match &self.0 {
27            Some(id) => id.fmt(f),
28            // TODO: Someone funny could set appSessionId to "None" string.

30        }
31    }
32}

34/// Display a topic hash in human-readable format by comparing with predefined topic types.
35/// If the topic matches a known event type, returns the human-readable name.
36/// Otherwise, returns the hash as a string.
37pub fn display_topic(topic: &B256) -> String {
38    match *topic {
39        t if t == golem_base_storage_entity_created() => {
40            "GolemBaseStorageEntityCreated".to_string()
41        }
42        t if t == golem_base_storage_entity_deleted() => {
43            "GolemBaseStorageEntityDeleted".to_string()
44        }
45        t if t == golem_base_storage_entity_updated() => {
46            "GolemBaseStorageEntityUpdated".to_string()
47        }
48        t if t == golem_base_storage_entity_ttl_extended() => {
49            "GolemBaseStorageEntityTTLExptended".to_string()
50        }
51        _ => format!("{topic}"),
52    }
53}

55/// Display a vector of FilterSet<B256> topics in human-readable format.
56/// Each FilterSet is displayed as a list of human-readable topic names.
57pub fn display_topics(topics: &[FilterSet<B256>]) -> String {
58    if topics.is_empty() {
59        return "[]".to_string();
60    }
61
62    let topic_strings: Vec<String> = topics
63        .iter()
64        .map(|filter_set| {
65            if filter_set.is_empty() {
66                "[]".to_string()
67            } else {
68                let inner_topics: Vec<String> = filter_set
69                    .iter()
70                    .map(|topic| display_topic(topic))
71                    .collect();
72                format!("[{}]", inner_topics.join(", "))
73            }
74        })
75        .collect();
76
77    format!("[{}]", topic_strings.join(", "))
78}
