33/// Contains all three nonce values for an account
34#[derive(Debug, Clone, derive_more::Display)]
35#[display("Last tracked nonce: {}, next pending nonce: {next_pending_nonce}, current account nonce: {account_nonce}", display_option(&last_used_nonce))]
36pub struct NonceInfo {
37    /// Last nonce used by SDK code (saved during previous call to process_transaction)
38    pub last_used_nonce: Option<u64>,
39    /// Next nonce including pending transactions
40    pub next_pending_nonce: u64,
41    /// Current nonce value from blockchain (represents next nonce after last confirmed transaction)
42    pub account_nonce: u64,
43}
44
45impl NonceInfo {
46    /// Picks the appropriate nonce for the next transaction and logs relevant information.
47    /// Returns the maximum of next_pending_nonce and (last_used_nonce + 1).
48    pub fn pick_nonce(&self) -> u64 {
49        log::info!("Nonce info: {self}");
50
51        let nonce = match self.last_used_nonce {
52            Some(last_used) => std::cmp::max(self.next_pending_nonce, last_used + 1),
53            None => self.next_pending_nonce,
54        };
55
56        let pending = self.pending_transactions_count();
57        if pending > 0 {
58            log::debug!("Still processing {pending} pending transactions");
59        }
60
61        if let Some(last_used) = self.last_used_nonce {
62            if (last_used + 1) != self.next_pending_nonce {
63                log::warn!("Last used nonce is not equal to next pending nonce. Probably transaction was sent externally.");
64            }
65        }
66
67        nonce
68    }
69
70    /// Returns the number of pending transactions for this account.
71    /// Returns the count of transactions that have been sent but not yet mined.
72    pub fn pending_transactions_count(&self) -> u64 {
73        let pending = self.next_pending_nonce as i64 - self.account_nonce as i64;
74        pending.max(0) as u64
75    }
76}

147    /// Signs a transaction request using the account's signer.
148    /// Returns the signed transaction ready for encoding and submission.
149    async fn sign_transaction(
150        &self,
151        tx: TransactionRequest,
152    ) -> anyhow::Result<Signed<EthereumTypedTransaction<TxEip4844Variant>>> {
153        let tx = tx.build_unsigned()?;
154        let bytes = tx.encoded_for_signing();
155
156        let signature = self.signer.sign(&bytes).await?;
157        Ok(tx.into_signed(signature))
158    }

160    /// Encodes a signed transaction to RLP bytes for network submission.
161    /// Also logs the transaction hash and attempts to decode and recover the signer for debugging.
162    fn encode_transaction(
163        &self,
164        signed: &Signed<EthereumTypedTransaction<TxEip4844Variant>>,
165    ) -> anyhow::Result<Vec<u8>> {
166        let mut encoded = Vec::new();
167        signed.eip2718_encode(&mut encoded);
168
169        log::trace!(
170            "RLP encoded transaction (hash: 0x{:x}): 0x{}",
171            signed.hash(),
172            hex::encode(&encoded)
173        );
174
175        // Decode the transaction for debugging purposes.
176        let decoded_tx = EthereumTxEnvelope::<TxEip4844>::decode(&mut &encoded[..])
177            .map_err(|e| anyhow!("Failed to decode transaction: {e}"))?;
178        log::debug!("Decoded transaction: {:#?}", decoded_tx);
179
180        let signer = decoded_tx
181            .recover_signer()
182            .map_err(|e| anyhow!("Failed to recover signer: {e}"))?;
183        log::debug!("Recovered signer: {:#?}", signer);
184
185        Ok(encoded)
186    }

204    /// Returns a new TransactionRequest with bumped tip and fee cap by a percentage for replacement transactions.
205    #[allow(unused)]
206    fn bump_fees(&self, request: &TransactionRequest, attempt: u32) -> TransactionRequest {
207        let bump_percent = self.tx_config.price_bump_percent * attempt as u128;
208        let tip = request
209            .max_priority_fee_per_gas
210            .unwrap_or(self.tx_config.max_priority_fee_per_gas);
211        let fee_cap = request
212            .max_fee_per_gas
213            .unwrap_or(self.tx_config.max_fee_per_gas);
214        let bumped_tip = tip + (tip * bump_percent).div_ceil(100);
215        let bumped_fee_cap = fee_cap + (fee_cap * bump_percent).div_ceil(100);
216        request
217            .clone()
218            .with_max_priority_fee_per_gas(bumped_tip)
219            .with_max_fee_per_gas(bumped_fee_cap)
220    }

222    /// Processes a single transaction:
223    /// - Gets the current nonce for the sender.
224    /// - Signs and encodes the transaction.
225    /// - Sends the transaction and waits for it to be mined.
226    /// - Returns the transaction receipt.
227    async fn process_transaction(&self, request: TransactionRequest) -> TransactionResponse {
228        // Get the current nonce for the sender address.
229        let from = request
230            .from
231            .ok_or_else(|| anyhow!("Transaction request missing 'from' address"))?;
232
233        // Wait for any pending transactions to be mined before proceeding.
234        // We allow only a single transaction to be pending at a time.
235        // If something goes wrong on blockchain and transaction can't pass, in most cases
236        // there is nothing we can do about it. We should avoid sending new transactions, because
237        // there will be more transactions stacked in the mempool.
238        // We shouldn't rather replace transactions, because bumping gas price introduces another
239        // layer of complexity.
240        self.wait_for_pending_transactions(from, self.tx_config.pending_transaction_timeout)
241            .await?;
242
243        // We have 2 sources of nonces: our last used nonce and RPC.
244        // RPC returns nonce of last pending transaction and nonce associated with the account.
245        // Since we have no guarantee of sending requests to the same RPC, we can't trust fully
246        // that it have full knowledge the returned nonces. At the same time tools outside of our
247        // control can send transactions as well.
248        let nonce_info = self.get_nonces(from).await?;
249        let nonce = nonce_info.pick_nonce();
250
251        // Update the request with the next pending nonce.
252        let mut request = request.with_nonce(nonce);
253
254        let max_retries = self.tx_config.max_retries;
255        let mut attempt: u32 = 0;
256
257        loop {
258            // Sign and encode the transaction.
259            let signed = self.sign_transaction(request.clone()).await?;
260            let encoded = self.encode_transaction(&signed)?;
261
262            attempt += 1;
263
264            let pending = match self.provider.send_raw_transaction(&encoded).await {
265                Ok(pending) => pending,
266                // Retry transaction with updated nonce.
267                Err(e) if e.to_string().contains("nonce too low") => {
268                    let nonce_info = self.get_nonces(from).await?;
269                    let nonce = nonce_info.pick_nonce();
270                    request.set_nonce(nonce);
271                    continue;
272                }
273                Err(e) => return Err(anyhow!("Failed to send transaction: {e}")),
274            };
275
276            let tx_hash = *pending.tx_hash();
277
278            log::debug!("Transaction attempt {attempt} sent with hash: {tx_hash}");
279
280            if let Some(receipt) = self.get_receipt_with_retry(tx_hash).await? {
281                log::info!("Transaction succeeded on attempt {attempt} with hash: {tx_hash}");
282
283                self.set_last_used_nonce(nonce);
284                return Ok(receipt);
285            }
286
287            if attempt >= max_retries {
288                return Err(anyhow!(
289                    "Transaction failed after {max_retries} attempts, last hash: {tx_hash}"
290                ));
291            }
292
293            log::warn!("Transaction attempt {attempt} timed out (hash: {tx_hash}), retrying...",);
294            tokio::time::sleep(Duration::from_millis(100)).await;
295        }
296    }

298    /// Spawns a worker task to process queued transactions in the background.
299    /// The worker receives transaction requests, processes them, and sends back receipts.
300    fn spawn_worker(mut rx: mpsc::Receiver<QueueMessage>, queue: Arc<Self>) {
301        let runtime = Builder::new_current_thread().enable_all().build().unwrap();
302
303        // We have 2 options to spawn signing worker task: using `spawn` or `spawn_local`.
304        // Using `spawn_local` can panic when called outside of LocalSet. That means that
305        // we force library consumer to use actix runtime or to manually create LocalSet.
306        // On the other hand using `spawn` will prevent consumer from using `spawn_local` in
307        // signing function.
308        // Spawning thread here might be overkill, but it's the only way to avoid affecting users.
309        std::thread::spawn(move || {
310            let local = LocalSet::new();
311
312            local.spawn_local(async move {
313                while let Some(msg) = rx.recv().await {
314                    let QueueMessage {
315                        request,
316                        response_tx,
317                    } = msg;
318
319                    let result = match tokio::time::timeout(
320                        queue.tx_config.transaction_receipt_timeout,
321                        queue.process_transaction(request),
322                    )
323                    .await
324                    {
325                        Ok(result) => result,
326                        Err(e) => Err(anyhow!(
327                            "Transaction processing timed out (timeout: {}): {e}",
328                            humantime::format_duration(queue.tx_config.transaction_receipt_timeout)
329                        )),
330                    };
331                    let _ = response_tx.send(result);
332                }
333            });
334
335            runtime.block_on(local);
336        });
337    }

339    /// Gets all three nonce values for this account.
340    /// Returns a NonceInfo struct containing last_used_nonce, next_pending_nonce, and current_blockchain_nonce.
341    async fn get_nonces(&self, address: Address) -> Result<NonceInfo> {
342        // Get last used nonce from stored value
343        let last_used_nonce = *self.last_used_nonce.lock().unwrap();
344
345        // Get current blockchain nonce from get_nonce. This function includes only
346        // transaction included in the latest block.
347        let account_nonce = self.provider.get_nonce(address).await?;
348
349        // Get next pending nonce from get_transaction_count. This function includes
350        // pending transactions as well.
351        let next_pending_nonce = self.provider.get_transaction_count(address).await?;
352
353        Ok(NonceInfo {
354            last_used_nonce,
355            next_pending_nonce,
356            account_nonce,
357        })
358    }

366    /// Queues a transaction for processing and returns a channel to await the result.
367    /// The transaction will be signed, sent, and the receipt returned asynchronously.
368    async fn queue_transaction(&self, request: TransactionRequest) -> Result<TransactionChannel> {
369        let (response_tx, response_rx) = oneshot::channel();
370        let msg = QueueMessage {
371            request,
372            response_tx,
373        };
374        self.sender
375            .send(msg)
376            .await
377            .map_err(|e| anyhow!("Failed to queue transaction: {}", e))?;
378        Ok(TransactionChannel { response_rx })
379    }

381    /// Checks if there are pending transactions and waits for them to be mined until timeout.
382    /// Returns an error if pending transactions won't be mined within the timeout.
383    pub async fn wait_for_pending_transactions(
384        &self,
385        address: Address,
386        timeout: Duration,
387    ) -> Result<()> {
388        let start_time = std::time::Instant::now();
389        let mut last_pending_count = 0u64;
390
391        loop {
392            let nonce_info = self.get_nonces(address).await?;
393
394            // Check if there are pending transactions using the same logic as pick_nonce
395            let pending = nonce_info.pending_transactions_count();
396            if pending == 0 {
397                log::debug!("No pending transactions found for address {address}");
398                return Ok(());
399            }
400
401            // Only log when the pending count changes
402            if pending != last_pending_count {
403                log::info!("Found {pending} pending transaction(s) (address {address}), waiting for them to be mined...");
404                last_pending_count = pending;
405            }
406
407            // Check if we've exceeded the timeout
408            if start_time.elapsed() >= timeout {
409                bail!("Timeout: {pending} transaction(s) are still pending (address {address})");
410            }
411
412            // Wait a bit before checking again
413            tokio::time::sleep(Duration::from_millis(500)).await;
414        }
415    }

522    /// Funds an account by sending ETH from a node-managed account.
523    /// This is typically used in development mode for test funding.
524    pub async fn fund_account(&self, value: BigDecimal) -> anyhow::Result<TransactionReceipt> {

541            .provider
542            .send_transaction(tx)

544            .map_err(|e| anyhow!("Failed to send transaction: {}", e))?;
545        self.transaction_queue
546            .get_receipt_with_retry(*pending.tx_hash())
547            .await?
548            .ok_or_else(|| anyhow!("Transaction receipt not found for funding transaction"))
549    }

558/// Gets a transaction receipt with retries for "transaction indexing is in progress" errors.
559/// Waits until the transaction is indexed and the receipt is available, or returns an error.
560/// If a timeout is provided and no receipt is received within that time, returns None.
561pub async fn get_receipt(
562    provider: &ResilientProvider,
563    tx_hash: Hash,
564    timeout_duration: Option<Duration>,
565    confirmations: u64,
566) -> anyhow::Result<Option<TransactionReceipt>> {
567    let start_time = std::time::Instant::now();
568    let _ = provider.wait_for_indexing(tx_hash, timeout_duration).await;
569
570    loop {
571        // Recalculate timeout in case it decreased during retries
572        let elapsed = start_time.elapsed();
573        let remaining_timeout = timeout_duration.map(|timeout| timeout.saturating_sub(elapsed));
574
575        if let Some(remaining_timeout) = remaining_timeout {
576            if remaining_timeout == Duration::ZERO {
577                return Ok(None);
578            }
579        }
580
581        let config = PendingTransactionConfig::new(tx_hash)
582            .with_required_confirmations(confirmations)
583            .with_timeout(remaining_timeout);
584        provider
585            .watch_for_confirmation(config)
586            .await
587            .map_err(|e| anyhow!("Failed watching for {tx_hash} confirmation: {e}"))?;
588
589        match provider
590            .get_transaction_receipt(tx_hash)
591            .await
592            .map_err(|e| anyhow!("Failed to get transaction receipt: {e}"))?
593        {
594            Some(receipt) => {
595                log::info!(
596                    "Transaction {tx_hash} was included in a block {:?} ({:?}). Waiting for {confirmations} confirmations.",
597                    receipt.block_number,
598                    receipt.block_hash.map(|b| hex::encode(b))
599                );
600                return Ok(Some(receipt));
601            }
602            None => {
603                log::trace!("Getting receipt returned None for transaction: {tx_hash}");
604                tokio::time::sleep(Duration::from_millis(200)).await;
605                continue;
606            }
607        }
608    }
