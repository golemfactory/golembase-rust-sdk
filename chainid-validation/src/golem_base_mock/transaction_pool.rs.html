1use crate::block::Transaction;
2use alloy::primitives::Address;
3use alloy::primitives::B256;
4use alloy::primitives::U256;
5use std::collections::HashMap;
6use std::sync::Arc;
7use tokio::sync::RwLock;
8
9/// Transaction pool that holds pending transactions
10#[derive(Clone, Debug, Default)]
11pub struct TransactionPool {
12    state: Arc<RwLock<HashMap<B256, Arc<Transaction>>>>,
13}
14
15impl TransactionPool {
16    /// Create a new empty transaction pool
17    pub fn new() -> Self {
18        Self {
19            state: Arc::new(RwLock::new(HashMap::new())),
20        }
21    }
22
23    /// Add a transaction to the pool
24    pub async fn add_transaction(&self, transaction: Arc<Transaction>) {
25        let hash = transaction.hash;
26        self.state.write().await.insert(hash, transaction);
27        log::info!("Transaction 0x{:x} added to pool", hash);
28    }
29
30    /// Get a transaction from the pool by hash
31    pub async fn get_transaction(&self, hash: &B256) -> Option<Arc<Transaction>> {
32        self.state.read().await.get(hash).cloned()
33    }
34
35    /// Remove a transaction from the pool (when it gets mined)
36    pub async fn remove_transaction(&self, hash: &B256) -> Option<Arc<Transaction>> {
37        self.state.write().await.remove(hash)
38    }
39
40    /// Get all pending transactions
41    pub async fn get_all_transactions(&self) -> Vec<Arc<Transaction>> {
42        self.state.read().await.values().cloned().collect()
43    }
44
45    /// Get a batch of transactions (for mining)
46    pub async fn get_transaction_batch(&self, max_count: usize) -> Vec<Arc<Transaction>> {
47        let transactions = self.state.read().await;
48        transactions.values().take(max_count).cloned().collect()
49    }
50
51    /// Get the number of pending transactions
52    pub async fn count(&self) -> usize {
53        self.state.read().await.len()
54    }
55
56    /// Check if the pool is empty
57    pub async fn is_empty(&self) -> bool {
58        self.state.read().await.is_empty()
59    }
60
61    /// Get transaction count for an address (nonce)
62    pub async fn get_transaction_count(&self, address: &Address) -> U256 {
63        // For now, return a simple count based on transactions in the pool
64        // In a real implementation, this would track nonces per address
65        let transactions = self.state.read().await;
66        let count = transactions
67            .values()
68            .filter(|tx| tx.from == *address)
69            .count();
70        U256::from(count)
71    }
72
73    /// Get transaction receipt (mock implementation)
74    pub async fn get_receipt(&self, _hash: &B256) -> Option<serde_json::Value> {
75        // Mock implementation - return None for now
76        // In a real implementation, this would return actual transaction receipts
77        None
78    }
79
80    /// Get transaction by hash
81    pub async fn get_transaction_by_hash(&self, hash: &B256) -> Option<Arc<Transaction>> {
82        // Return the actual transaction object
83        self.state.read().await.get(hash).cloned()
84    }
85}
